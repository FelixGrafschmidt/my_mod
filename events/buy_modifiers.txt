	}
	option = {

		ai_chance = {
			factor = 100
		}
		country_event = {
			id = buy_modifiers.0
		}
	}
	option = {
		name = buy_modifiers.1.c
		add_country_modifier = {
			name = buy_modifiers_religion_1

		}

			}
		}

		country_event = {
			id = buy_modifiers.1
		}
	}
}
